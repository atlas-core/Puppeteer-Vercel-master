const sqlite3 = require('sqlite3').verbose();

// Create a new SQLite database instance
const db = new sqlite3.Database('example.db');

// Run a CREATE TABLE statement to create a new table
db.serialize(() => {
  db.run(`
    CREATE TABLE IF NOT EXISTS users (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      name TEXT,
      email TEXT
    )
  `);

  console.log('Table created successfully');

  // Insert a sample record into the table
  db.run(`
    INSERT INTO users (name, email) VALUES (?, ?)
  `, ['John Doe', 'john@example.com'], function(err) {
    if (err) {
      console.error('Error inserting record:', err);
    } else {
      console.log(`Record inserted with ID: ${this.lastID}`);
    }
  });

  // Retrieve all records from the table
  db.all('SELECT * FROM users', (err, rows) => {
    if (err) {
      console.error('Error retrieving records:', err);
    } else {
      console.log('Records retrieved successfully');
      console.log(rows);
    }
  });
});

// Close the database connection
db.close();
